"""Simulate leaflet.heat math to check visibility."""
import math

# Current: backend scores 3-37, frontend passes as-is, leaflet.heat max=1
# leaflet.heat internally does: alpha = intensity / max * radius_factor
# If max=1 and intensity=33, then alpha=33 (clamped to 1 if no radius factor)
# But with radius factor (~0.000064), alpha=0.0021 — invisible!

# The fix: normalize weights to 0-1 range by dividing by 100
# Then with max=1, alpha = weight / 1 = 0.03-0.37 — visible!

print("CURRENT STATE:")
print("  Backend scores: 3-37")
print("  Frontend weight: passes score as-is")
print("  leaflet.heat max: 1")
print("  Result: weights 3-37 with max=1, simpleheat formula uses intensity/max,")
print("          so alpha=3 to 37, clamped to 1. SHOULD be visible (solid purple).")
print()

print("But if leaflet.heat adds a radius divisor to intensity:")
print("  f = 1/(2*PI*r^2) where r=radius*2=50")
print("  f = 1/(2*3.14159*2500) = 0.000064")
print("  alpha = min(1, max(0, 33 * 0.000064 / 1)) = 0.0021")
print("  Alpha of 0.0021 is INVISIBLE on a dark map!")
print()

print("ROOT CAUSE: weight values (3-37) are not normalized to 0-1 range")
print("while leaflet.heat expects intensities that normalize correctly")
print("with its internal radius-based calculation.")
print()

print("RECOMMENDED FIX:")
print("  Divide weight by 100 in frontend api.ts mapping")
print("  This gives weights 0.03-0.37, which works with leaflet.heat")
print()
print("  Also set leaflet.heat max to be auto-computed or omit it to")
print("  let the library use the max value from the data (or 1 if not set).")
